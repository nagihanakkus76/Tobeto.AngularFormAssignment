Angular Built-In && Custom Pipes && Directives

  1. Date Pipe
    {{date}} - {{date | date:'short'}}
    {{date}} - {{date | date:'fullDate'}}
    {{date}} - {{date | date:'medium'}}
    {{date}} - {{date | date:'mm:ss'}}

  2. UpperCase Pipe
    {{title}} - {{title | uppercase}}

  3. LowerCasePipe
    {{title | uppercase}} - {{title | lowercase}}

  4. Currency Pipe
    {{num}} - {{num | currency}}

  5. Percent Pipe
    {{num}} - {{num | percent}}

  6. Decimal Pipe
    {{num2}} - {{num2 | number:'1.2-2'}}

  7. Slice Pipe
    {{title}} - {{title | slice:7:12}}

  8. TitleCase Pipe
    {{title}} - {{title | titlecase}}

  9. Json Pipe
    {{title}} - {{title | json}}
    {{num}} - {{num | json}}
